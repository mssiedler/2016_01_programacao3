package dao;

import java.util.List;
import javax.persistence.EntityManager;
import javax.persistence.EntityManagerFactory;
import modelo.${Classe};

/**
 *
 * @author marcelosiedler
 */
public class ${Classe}DAO {
    EntityManager em;
    
    public ${Classe}DAO() throws Exception {
        EntityManagerFactory emf;
        emf = Conexao.getConexao();
        em =  emf.createEntityManager();
    }
    
    public void incluir(${Classe} obj) throws Exception {
        try {
            em.getTransaction().begin();
            em.persist(obj);
            em.getTransaction().commit();
        } catch (RuntimeException e) {
            em.getTransaction().rollback();
            throw e;
        } finally {
            //em.close();
            
        }
        
    }
    public List<${Classe}> listar() throws Exception {
        return em.createNamedQuery("${Classe}.findAll").getResultList();
    }
    
    public void alterar(${Classe} obj) throws Exception {
        
        try {
            em.getTransaction().begin();
            em.merge(obj);
            em.getTransaction().commit();
        } catch (RuntimeException e) {
            em.getTransaction().rollback();
            throw e;
        } finally {
           // em.close();
        }
    }
    
    public void excluir(${Classe} obj) throws Exception {
        
        try {
            em.getTransaction().begin();
            em.remove(obj);
            em.getTransaction().commit();
        } catch (RuntimeException e) {
            em.getTransaction().rollback();
        } finally {
            //em.close();
        }
    }
    public void fechaEmf() {
        em.close();
        Conexao.closeConexao();
    }






    

}
